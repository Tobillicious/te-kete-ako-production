```sql
WHERE status = 'working'

-- 4. Update your status
UPDATE agent_status
SET status = 'working',
    current_task = '[task description]',
    files_editing = ARRAY['file1.html', 'file2.js'],
    last_heartbeat = NOW()
WHERE agent_id = '[your-id]';
```

### **Every Hour - Heartbeat:**

```sql

SET last_heartbeat = NOW()
WHERE agent_id = '[your-id]';
```
