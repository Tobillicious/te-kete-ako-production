#### **Enhancement 4: Vision Tracking View**

**Create:**
```sql
CREATE OR REPLACE VIEW complete_vision_network AS
SELECT 
  'Planning' as layer,
  r.title,
  r.type,
  r.created_at::date,
  (SELECT COUNT(*) FROM graphrag_relationships WHERE source_resource_id = r.id) as outgoing_links,
  (SELECT COUNT(*) FROM graphrag_relationships WHERE target_resource_id = r.id) as incoming_links
FROM resources r
WHERE r.type IN ('Strategic Vision', 'Cultural Design Philosophy', 'UX Specification', 'Production Timeline')

UNION ALL

SELECT 
  'Product' as layer,
  r.title,
  r.type,
  r.created_at::date,
  (SELECT COUNT(*) FROM graphrag_relationships WHERE source_resource_id = r.id) as outgoing_links,
  (SELECT COUNT(*) FROM graphrag_relationships WHERE target_resource_id = r.id) as incoming_links
FROM resources r
WHERE r.type IN ('Dashboard Page', 'Serverless Function', 'UI Component', 'Intelligence Tool')

ORDER BY layer, created_at;
```
